24 partial void AfterExecuteInsert(TEntity entity)

36 partial void BeforeExecuteUpdate(TEntity entity)
37 {
38 var historyItem = historyItems.FirstOrDefault(e => e.Entity.Id == entity.Id);
39
40 if (historyItem == null)
41 {
42 var oriEntity = EntitySet.Where(e => e.Id == entity.Id).AsNoTracking().FirstOrDefault();
43
44 if (oriEntity != null)
45 {
46 var loginSession = Modules.Account.AccountManager.QueryLoginSession(SessionToken);
47
48 if (loginSession != null)
49 {
50 historyItems.Add(new HistoryItem(loginSession.IdentityId, RevisionType.Update, DateTime.Now, entity, JsonSerializer.Serialize(oriEntity, new JsonSerializerOptions() { MaxDepth = 0, })));
51 }
52 }
53 }
54 }
55 partial void BeforeExecuteDelete(TEntity entity)
56 {
57 var loginSession = Modules.Account.AccountManager.QueryLoginSession(SessionToken);
58
59 if (loginSession != null)
60 {
61 historyItems.Add(new HistoryItem(loginSession.IdentityId, RevisionType.Delete, DateTime.Now, entity, JsonSerializer.Serialize(entity, new JsonSerializerOptions() { MaxDepth = 0 })));
62 }
63 }
64 partial void AfterExecuteSaveChanges()
65 {
66 if (historyItems.Any())
67 {
68 Task.Run(async () =>
69 {
70 using var ctrl = new Revision.HistoriesController();
71
72 ctrl.SessionToken = Modules.Security.Authorization.SystemAuthorizationToken;
73 foreach (var item in historyItems)
74 {
75 var history = new Entities.Revision.History()
76 {
77 IdentityId = item.IdentityId,
78 ActionType = item.RevisionType.ToString(),
79 ActionTime = item.ActionTime,
80 SubjectName = item.Entity.GetType().Name,
81 SubjectId = item.Entity.Id,
82 JsonData = item.JsonData ?? string.Empty,
83 };
84 _ = await ctrl.ExecuteInsertAsync(history).ConfigureAwait(false);
85 }
86 _ = await ctrl.SaveChangesAsync().ConfigureAwait(false);
